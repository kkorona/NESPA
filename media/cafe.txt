#include <iostream>
#include<utility>
#include <vector>
#include<algorithm>
using namespace std;

class cafe{public:int cur,pre,next;};

int pre_point(bool *visit, int val, int n){
    int cur=val-1;
    while( cur!=val ){//한바퀴 까지
        if(visit[cur]==true)
            break;
        if(cur==0){
            cur=n;//다시 돌아가서 감소
            continue;

        }
        --cur;
    }
    return cur;//cur==val
}
int next_point(bool *visit, int val, int n){
    int cur=val+1;
    while( cur!=val ){
        if(visit[cur]==true)
            break;
        if(cur==n){
            cur=0;
            continue;

        }
        ++cur;
    }
    return cur;//cur==val
}

int main(){
    int n,k;

    cin>>n>>k;
    cafe c[n+1];
    bool visit[n+1]={false,};
    int pr,ne;
    int idx=0;
    int val=0;
    for(int i=0; i<k; i++){
        cin>>val ; //좌석 입장 4, 9, 15, 6, 22, 3, 3, 17
        if(visit[val]){//이미 방문했다면?->퇴장
            visit[val]=false;// 카페에서 입장상태가 아닌것으로
            //--idx;
            continue;
        }
        visit[val]=true;//방문
        pr = pre_point( visit,val,n);//이전 위치 탐색
        ne = next_point(visit,val,n);//다음 위치 탐색
        c[idx++] = {val, pr, ne};

       // cout<<c[i].n<<endl;
       //

    }


    for(int i=0; i<idx; i++)
        cout<<c[i].cur<<" "<<c[i].pre<<" "<<c[i].next<<endl;
    return 0;
}
